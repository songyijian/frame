绑定数据 

创建一组对象：.selectAll('')
数据绑定 ：.data(data)-绑定数据 
循环: .enter()


let rectList = osvg.selectAll('rect')
    .data(datas) // ==vue data  ==react getinitalstate
    .enter() 
    .append('rect')
    .style('fill', 'red')
    .attr('width', function(n) {   
        console.log(n);
        return n * 2;
    })
    .attr('height', 50)
    .attr('y', (n, index) => {  // == forEach(item, index)
        console.log(n, index)
        return index * 60;
    })
